50 template <typename FloatType, typename WinGenType>
51 std::vector<FloatType> FirLowPassFilter(size_t numTaps, FloatType cutoffFreqHz,
52  FloatType samplingFreqHz, WinGenType const& windowGenerator)
53 {
54  DSP_ASSERT_THROW(numTaps > 2, "numTaps too small");
55  DSP_ASSERT_THROW(cutoffFreqHz > 0, "cutoffFreqHz <= 0");
56  DSP_ASSERT_THROW(samplingFreqHz > 0, "samplingFreqHz <= 0");
57  auto availableBandwidthHz = samplingFreqHz / 2;
58  DSP_ASSERT_THROW(cutoffFreqHz <= availableBandwidthHz, "cutoffFreqHz too high");
59  static const auto PI = Pi<FloatType>();
60  std::vector<FloatType> filterCoeffs(numTaps);
61  auto normalizedCutoffFreq = cutoffFreqHz / availableBandwidthHz;
62  auto numTapsMinusOneOverTwo = static_cast<FloatType>(numTaps - 1) / 2;
63  size_t i = 0;
64 
65  for (auto& filterCoeff : filterCoeffs)
66  {
67  auto arg = static_cast<FloatType>(i) - numTapsMinusOneOverTwo;
68  filterCoeff = normalizedCutoffFreq * Sinc(normalizedCutoffFreq * arg * PI);
69  ++i;
70  }
71 
72  WindowFunction<FloatType> window(windowGenerator, numTaps, false);
73  window(filterCoeffs.begin(), filterCoeffs.end(), filterCoeffs.begin());
74  return filterCoeffs;
75 }

133 template <typename FloatType, typename WinGenType>
134 std::vector<FloatType> FirBandPassFilter(size_t numTaps, FloatType centreFreqHz,
135  FloatType bandwidthHz, FloatType samplingFreqHz,
136  WinGenType const& windowGenerator)
137 {
138  DSP_ASSERT_THROW(numTaps > 2, "numTaps too small");
139  DSP_ASSERT_THROW(centreFreqHz > 0, "cutoffFreqHz <= 0");
140  DSP_ASSERT_THROW(samplingFreqHz > 0, "samplingFreqHz <= 0");
141  auto availableBandwidthHz = samplingFreqHz / 2;
142  DSP_ASSERT_THROW(centreFreqHz <= availableBandwidthHz, "cutoffFreqHz too high");
143  DSP_ASSERT_THROW(bandwidthHz > 0, "bandwidthHz <= 0");
144  DSP_ASSERT_THROW(bandwidthHz <= availableBandwidthHz, "bandwidthHz too high");
145  static const auto PI = Pi<FloatType>();
146  std::vector<FloatType> filterCoeffs(numTaps);
147  auto normalizedCentreFreq = centreFreqHz / availableBandwidthHz;
148  auto normalizedBandwidth = bandwidthHz / availableBandwidthHz;
149  auto normalizedLowCutoffFreq = normalizedCentreFreq - (normalizedBandwidth / 2);
150  auto normalizeHighCutoffFreq = normalizedCentreFreq + (normalizedBandwidth / 2);
151  auto numTapsMinusOneOverTwo = static_cast<FloatType>(numTaps - 1) / 2;
152  size_t i = 0;
153 
154  for (auto& filterCoeff : filterCoeffs)
155  {
156  auto arg = static_cast<FloatType>(i) - numTapsMinusOneOverTwo;
157  filterCoeff = std::abs(arg) < FloatType(1.e-3) ? FloatType(0.)
158  : (cos(normalizedLowCutoffFreq * arg * PI) -
159  cos(normalizeHighCutoffFreq * arg * PI)) /
160  PI / arg;
161  ++i;
162  }
163 
164  WindowFunction<FloatType> window(windowGenerator, numTaps, false);
165  window(filterCoeffs.begin(), filterCoeffs.end(), filterCoeffs.begin());
166  return filterCoeffs;
167 }

283  template <typename IterIn, typename OutIter>
284  void operator()(IterIn signalFirst, IterIn signalLast, OutIter resultFirst, bool removeDelay)
285  {
286  auto signalLen = std::distance(signalFirst, signalLast);
287  DSP_ASSERT_THROW(signalLen == static_cast<decltype(signalLen)>(m_signalLength),
288  "signal sample range incorrect");
289 
290  if (m_useFastConvolution)
291  {
292  m_fftConvolve(signalFirst,
293  signalLast,
294  m_filterCoeffs.begin(),
295  m_filterCoeffs.end(),
296  m_filteredSignal.begin());
297  }
298  else
299  {
300 
301  Convolve(signalFirst,
302  signalLast,
303  m_filterCoeffs.begin(),
304  m_filterCoeffs.end(),
305  m_filteredSignal.begin());
306  }
307 
308  if (removeDelay)
309  {
310  auto offset = m_filteredSignal.size() - m_signalLength;
311  offset >>= 1;
312  auto first = std::next(m_filteredSignal.begin(), static_cast<int>(offset));
313  auto last = std::next(first, signalLen);
314 
315  std::copy(first, last, resultFirst);
316  }
317  }
