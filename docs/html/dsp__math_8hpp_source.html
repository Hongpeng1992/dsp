60 template <typename InIter1, typename InIter2, typename OutIter>
61 void Convolve(InIter1 first1, InIter1 last1, InIter2 first2, InIter2 last2, OutIter result)
62 {
63  using value_t = typename std::iterator_traits<OutIter>::value_type;
64 
65  auto M = std::distance(first1, last1);
66  DSP_ASSERT_THROW(M > 0, "range 1 invalid");
67 
68  auto N = std::distance(first2, last2);
69  DSP_ASSERT_THROW(N > 0, "range 2 invalid");
70 
71  auto K = M + N - 1;
72 
73  for (auto k = 0; k < K; ++k)
74  {
75  *result = value_t(0);
76  auto kMin = k >= N - 1 ? k - (N - 1) : 0;
77  auto kMax = k < M - 1 ? k : M - 1;
78 
79  for (auto kSub = kMin; kSub <= kMax; ++kSub)
80  {
81  *result +=
82  static_cast<value_t>(*std::next(first1, kSub) * *std::next(first2, k - kSub));
83  }
84 
85  std::advance(result, 1);
86  }
87 }

97 {
98  static_assert(std::is_floating_point<FloatType>::value, "invalid floating point type");
99 
100  FloatType sum{0};

103  {
104  auto xToIpower = std::pow(x / 2, static_cast<FloatType>(i));
105  int factorial = 1;
