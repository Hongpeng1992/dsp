46 template <typename FloatType, typename EvalFunc>
47 void WindowGenerator(std::vector<FloatType>& windowCoeffs, EvalFunc evalCoeff)
48 {
49  auto size = windowCoeffs.size();
50  DSP_ASSERT_THROW(size > 1, "invalid window size");
51  auto sizeMinusOne = static_cast<FloatType>(size - 1);
52  auto halfSize = size >> 1;
53 
54  for (size_t n = 0, nRev = size - 1; n < halfSize; ++n, --nRev)
55  {
56  windowCoeffs[n] = evalCoeff(static_cast<FloatType>(n), sizeMinusOne);
57  windowCoeffs[nRev] = windowCoeffs[n];
58  }
59 
60  if (1 == size % 2)
61  {
62  windowCoeffs[halfSize] = evalCoeff(static_cast<FloatType>(halfSize), sizeMinusOne);
63  }
64 }

82 template <typename FloatType, typename Iter>
83 FloatType EvaluateFlatTopCoefficient(FloatType n, FloatType sizeMinusOne, Iter eqCoeffsFirst,
84  Iter eqCoeffsLast)
85 {
86  static const auto TWO_PI = TwoPi<FloatType>();
87  auto numEqCoeffs = std::distance(eqCoeffsFirst, eqCoeffsLast);
88  DSP_ASSERT_THROW(numEqCoeffs > 1, "invalid number of equation coefficients");
89  auto twoPiN = TWO_PI * n;
90  auto sign = static_cast<FloatType>(-1);
91  auto windowCoeff = *eqCoeffsFirst;
92  int i = 1;
93 
94  for (auto itr = std::next(eqCoeffsFirst); itr != eqCoeffsLast;
95  std::advance(itr, 1), sign *= static_cast<FloatType>(-1), ++i)
96  {
97  windowCoeff += sign * *itr * std::cos((static_cast<FloatType>(i) * twoPiN) / sizeMinusOne);
98  }
99 
100  return windowCoeff;
101 }

641  template <typename Generator>
642  WindowFunction(Generator const& generator, size_t size, bool ignoreLastValue)
643  : m_windowCoefficients(size, static_cast<FloatType>(1))

701  template <typename InIter, typename OutIter>
702  static void ApplyGainCorrection(InIter dataFirst, InIter dataLast, OutIter resultFirst,
703  FloatType gain)
704  {
705  for (auto itr = dataFirst; itr != dataLast;
706  std::advance(itr, 1), std::advance(resultFirst, 1))
707  {
708  *resultFirst = *itr / gain;
709  }

769  template <typename InIter, typename OutIter>
770  void operator()(InIter dataFirst, InIter dataLast, OutIter resultFirst) const
771  {
772  auto inputLength = std::distance(dataFirst, dataLast);
773  DSP_ASSERT_THROW(inputLength == static_cast<decltype(inputLength)>(m_effectiveSize),
774  "invalid data size");
775 
776  // Generic lambda function to allow multiplying mixed types.
777  // This is because we may want to window complex data not just
778  // real valued data.
779  auto multiply = [](auto const& x, auto const& y) { return x * y; };
780 
781  // Transform the data by applying the window coefficients using
782  // piecewise multiplication.
783  std::transform(dataFirst, dataLast, m_windowCoefficients.begin(), resultFirst, multiply);
784  }

786 private:
788  void ComputeGains()
789  {
790  auto size = EffectiveSize();
791 
792  for (size_t i = 0; i < size; ++i)
793  {
794  m_coherentGain += m_windowCoefficients[i];
795  m_enbw += m_windowCoefficients[i] * m_windowCoefficients[i];
796  }
797 
798  auto enbwDivisor = m_coherentGain * m_coherentGain;
799 
800  if (std::abs(enbwDivisor) > static_cast<FloatType>(1.e-9))
801  {
802  m_enbw = (static_cast<FloatType>(size) * m_enbw) / enbwDivisor;
803  }
804 
805  m_coherentGain /= static_cast<FloatType>(size);
806  m_powerGain = m_coherentGain * m_coherentGain * m_enbw;
807  }

813  bool m_ignoreLastValue{false};
815  size_t m_effectiveSize{0};
817  FloatType m_coherentGain{0.};
819  FloatType m_powerGain{0.};
821  FloatType m_enbw{0.};
822 };
