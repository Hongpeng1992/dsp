50 std::pair<size_t, size_t> ComputeResampleFactors(double requiredResampleFactor,
51  size_t maxNumerator = 128,
52  size_t maxDenominator = 128)
53 {
54  DSP_ASSERT_THROW(requiredResampleFactor > 0, "requiredResampleFactor <= 0");
55  std::pair<size_t, size_t> factors(0, 0);
56  auto n_a = static_cast<size_t>(std::floor(requiredResampleFactor) + 0.5);
57  size_t d_a = 1;
58  auto n_b = static_cast<size_t>(std::ceil(requiredResampleFactor) + 0.5);
59  size_t d_b = 1;
60  double error = std::numeric_limits<double>::max();
61 
62  while (true)
63  {
64  auto n_m = n_a + n_b;
65  auto d_m = d_a + d_b;
66 
67  // Make sure we reduce the numerator and denominator as much as possible.
68  // Although in practice we may never need this here as n_m and d_m are
69  // likely already in their lowest form.
70  auto g = Gcd(n_m, d_m);
71 
72  if (g > 1)
73  {
74  n_m /= g;
75  d_m /= g;
76  }
77 
78  if ((n_m > maxNumerator) || (d_m > maxDenominator))
79  {
80  break;
81  }
82 
83  double m = static_cast<double>(n_m) / static_cast<double>(d_m);
84  auto absDiff = std::abs(m - requiredResampleFactor);
85 
86  if (absDiff < error)
87  {
88  error = absDiff;
89  factors = {n_m, d_m};
90  }
91 
92  if (m <= requiredResampleFactor)
93  {
94  n_a = n_m;
95  d_a = d_m;
96  }
97  else
98  {
99  n_b = n_m;
100  d_b = d_m;
101  }
102  }
103 
104  return factors;
105 }

258  template <typename IterIn, typename OutIter>
259  void operator()(IterIn signalFirst, IterIn signalLast, OutIter resultFirst)
260  {
261  auto signalLength = std::distance(signalFirst, signalLast);
262  DSP_ASSERT_THROW(signalLength == static_cast<decltype(signalLength)>(m_signalLength),
263  "sample length is incorrect");
264 
265  if (m_upsampleFactor > 1)
266  {
267  // Make sure upsample buffer starts zeroed.
268  std::fill(
269  m_workspaceBuffer.begin(), m_workspaceBuffer.end(), static_cast<FloatType>(0));
270 
271  // Fill upsample buffer with signal samples with zero padding.
272  auto upIter = m_workspaceBuffer.begin();
273 
274  for (auto itr = signalFirst; (itr != signalLast) && (upIter < m_workspaceBuffer.end());
275  std::advance(itr, 1), std::advance(upIter, static_cast<int>(m_upsampleFactor)))
276  {
277  // Note: we correct for attenuation by upsample factor, caused by inserting zeroes.
278  *upIter = *itr * static_cast<FloatType>(m_upsampleFactor);
279  }
280 
281  // Low pass filter the workspace buffer.
282  m_filterHolder(m_workspaceBuffer.begin(),
283  m_workspaceBuffer.end(),
284  m_workspaceBuffer.begin(),
285  true);
286 
287  // If required perform downsampling.
288  if (m_downsampleFactor > 1)
289  {
290  // Fill the result buffer by dropping samples as required.
291  auto upIter = m_workspaceBuffer.begin();
292  auto resultIter = resultFirst;
293 
294  for (auto i = 0; i < m_resampledLength; ++i,
295  std::advance(upIter, static_cast<int>(m_downsampleFactor)),
296  std::advance(resultIter, 1))
297  {
298  *resultIter = *upIter;
299  }
300  }
301  // Else return upsample buffer as is copied ito the result buffer.
302  else
303  {
304  std::copy(m_workspaceBuffer.begin(), m_workspaceBuffer.end(), resultFirst);
305  }
306  }
307  else
308  {
309  // Low pass filter the signal data.
310  m_filterHolder(signalFirst, signalLast, m_workspaceBuffer.begin(), true);
311 
312  // Fill the result buffer by dropping samples as required.
313  auto resultIter = resultFirst;
314 
315  for (auto itr = m_workspaceBuffer.begin(); itr < m_workspaceBuffer.end();
316  std::advance(itr, static_cast<int>(m_downsampleFactor)),
317  std::advance(resultIter, 1))
318  {
319  *resultIter = *itr;
320  }
321  }
322  }

340  m_workspaceBuffer.resize(upsampleLength);
341  }
342  else

344  m_workspaceBuffer.resize(m_signalLength);
345  }
346  }
